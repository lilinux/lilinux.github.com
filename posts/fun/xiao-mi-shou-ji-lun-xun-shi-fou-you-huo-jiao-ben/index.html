然后我马上写了一个写完就扔的脚本

#!/usr/bin/env python
#-*-coding: utf-8 -*-

import time
import urllib2

url ='http://www.mi.com/buyphone/mi3'
#proxy = 'http://proxy.xxoo.com:8080'
proxy = None

if proxy:
    ph = urllib2.ProxyHandler({'http': proxy})
    opener=urllib2.build_opener(ph)
    urllib2.install_opener(opener)


def get_html(url=url):
    return urllib2.urlopen(url).read()


def line_value(line):
    return line[line.find('"') + 1 :line.rfind('"')]


def parse(html):
    result = []
    op = size = color = 'all'
    for line in html.split('\n'):
        if line.find('title:') > 0:
            value = line_value(line)
            if value.endswith('版'):
                op = value
                size = color = 'all'
            elif value.endswith('GB'):
                size = value
                color = 'all'
            else:
                color = value
        if line.find('isLack:') > 0:
            is_lack = line.find('true') > 0
            result.append(('-'.join((op, size, color)), is_lack))
    return result


def is_ok(result):
    d = dict(result)
    return not d['电信版-all-all'] and not d['电信版-16GB-all']


def notify():
    import os
    os.system('start chrome %s' % url)


def loop():
    while 1:
        print time.ctime()
        result = parse(get_html())
        for i, j in result:
            print i, '\t\t', j
        if is_ok(result):
            notify()
            break
        time.sleep(60)

if __name__ == '__main__':
    loop()

loop是主轮询逻辑，每分钟查询一次。获取到页面后分析是否有货，如果有货就调用notify通知。

脚本编写一共花了约30分钟，先让它跑两天看看。虽然有很多不完善的地方，比如notify和parse都很weak。
